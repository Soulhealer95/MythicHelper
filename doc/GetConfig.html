def getconf
  conf = nil
  if @type.include?("DB")
    conf = getDBConf()
  elsif @type.include?("Blizzard")
    conf = getBlizzConf()
  elsif @type.include?("Bot")
    conf = getDiscConf()
  else
    puts "Invalid config"
    return conf
  end
end
